CREATE TABLE Admin (
    admin_ID INT AUTO_INCREMENT PRIMARY KEY,
    admin_email VARCHAR(255) NOT NULL,
    admin_password VARCHAR(255) NOT NULL,
    verification_code VARCHAR(6),
    code_expiration TIMESTAMP
);

CREATE TABLE Content (
    content_ID INT AUTO_INCREMENT PRIMARY KEY,
    admin_ID INT,
    FOREIGN KEY (admin_ID) REFERENCES Admin(admin_ID),
    content_text TEXT,
    content_image VARCHAR(255)
);

CREATE TABLE Category (
    category_ID INT AUTO_INCREMENT PRIMARY KEY,
    category_name VARCHAR(255) NULL,
    category_image VARCHAR(255) NULL
);

CREATE TABLE Subcategory (
    subcategory_ID INT AUTO_INCREMENT PRIMARY KEY,
    subcategory_name VARCHAR(255) NOT NULL,
    category_ID INT,
    FOREIGN KEY (category_ID) REFERENCES Category(category_ID)
);

CREATE TABLE Product (
    product_ID INT AUTO_INCREMENT PRIMARY KEY,
    admin_ID INT,
    FOREIGN KEY (admin_ID) REFERENCES Admin(admin_ID),
    product_name VARCHAR(255) NOT NULL,
    category_ID INT,
    FOREIGN KEY (category_ID) REFERENCES Category(category_ID),
    subcategory_ID INT,
    FOREIGN KEY (subcategory_ID) REFERENCES Subcategory(subcategory_ID),
    product_description VARCHAR(255) NOT NULL,
    product_image VARCHAR(255) NOT NULL,
    product_price DECIMAL(10, 2) NOT NULL,
    product_popular_value BOOLEAN NOT NULL DEFAULT 0
);

CREATE TABLE ContactUs (
    contact_ID INT AUTO_INCREMENT PRIMARY KEY,
    admin_ID INT,
    FOREIGN KEY (admin_ID) REFERENCES Admin(admin_ID),
    contact_first_name VARCHAR(255) NULL,
    contact_last_name VARCHAR(255) NULL,
    contact_email VARCHAR(255) NOT NULL,
    contact_subject VARCHAR(255) NOT NULL,
    contact_message TEXT NOT NULL,
    contact_respond_value BOOLEAN NOT NULL DEFAULT 0,
    contact_timestamp TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE Customer (
    customer_ID INT AUTO_INCREMENT PRIMARY KEY,
    customer_email VARCHAR(255),
    customer_first_name VARCHAR(255),
    customer_last_name VARCHAR(255)
);

CREATE TABLE Feedback (
    feedback_ID INT AUTO_INCREMENT PRIMARY KEY,
    customer_ID INT,
    FOREIGN KEY (customer_ID) REFERENCES Customer(customer_ID),
    feedback_timestamp TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    feedback_experience VARCHAR(1000)
);

CREATE TABLE Section (
    section_ID INT AUTO_INCREMENT PRIMARY KEY,
    section_name VARCHAR(255) NOT NULL
);

CREATE TABLE Question (
    question_ID INT AUTO_INCREMENT PRIMARY KEY,
    section_ID INT,
    FOREIGN KEY (section_ID) REFERENCES Section(section_ID),
    question_text VARCHAR(255) NOT NULL,
    archive_value BOOLEAN NOT NULL DEFAULT 0
);

CREATE TABLE Rating (
    rating_ID INT AUTO_INCREMENT PRIMARY KEY,
    admin_ID INT,
    FOREIGN KEY (admin_ID) REFERENCES Admin(admin_ID),
    feedback_ID INT,
    FOREIGN KEY (feedback_ID) REFERENCES Feedback(feedback_ID),
    question_ID INT,
    FOREIGN KEY (question_ID) REFERENCES Question(question_ID),
    rating_number INT NOT NULL
);
